    public static  int[] insertion_sort( int[] array )
    {
        for (int i = 1; i < array.length; i++) {
            int j = i;
            int B = array[i];
            while (j > 0 && array[j - 1] > B) {

                j--;
            }
            array[j] = B;
        }
        return array;
